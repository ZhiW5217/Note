## 数学函数
```python
#pi
print(math.pi)

#e
print(math.e)

#正弦 余弦
print(math.sin(90))

#弧度转角度
print(math.degrees(2 * math.pi))

#角度转弧度

print(math.radians(180))
print(math.sin(math.radians(90)))

#求模

print(math.fmod(10, 3))

#上下取整

print(math.ceil(-10.2)) #上
print(math.floor(-10.2))#下


#求绝对值

print(math.fabs(-1))
print(math.fabs(-5))
print(math.fabs(-5.4))


#求开方数

print(math.sqrt(2))

#求次方

print(math.pow(10, 6))

#求对数

print(math.log(27, 3))

#求阶乘

print(math.factorial(100))


#整数的截取

print(math.trunc(10.1231))
print(math.trunc(-10.1231))
print(int(10.1231))
print(int(-10.1231))
```
